cmake_minimum_required(VERSION 3.1)
project(lib)

set(CMAKE_CXX_STANDARD 14)

set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -D_GLIBCXX_DEBUG -fsanitize=address,undefined")
set(CMAKE_CXX_FLAGS_RELEASE "-O2")

add_library(static STATIC static.cpp)
add_library(shared SHARED shared.cpp)
add_library(second_shared SHARED second_shared.cpp)

add_executable(lib libs.cpp)

TARGET_LINK_LIBRARIES(lib -ldl static shared)
